        public object Find(string name,object node)
        {
        object obj;
        if (node is Country country && country.is_checked == false)

               obj=Find(name,country.neibor2);
               if (obj != null) /*country.is_checked = false;*/ return obj;

                //country.is_checked = false;
                return null;
                   
            }

        if(node is City city && city.is_checked == false)
            {
                city.is_checked = true;

                if(city.name==name) /*city.is_checked = false;*/ return city;

                obj=Find(name,city.location);

                if (obj != null)/* city.is_checked = false;*/ return obj;

                //city.is_checked = false;
                return null;
            }


            if (node is Location location && location.is_checked == false)
            {
                location.is_checked = true;


                obj = Find(name, location.country);
                if (obj != null) /*location.is_checked = false;*/ return obj;
